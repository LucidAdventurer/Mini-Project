-- Enhanced Database Schema for PTA (Placement Training & Assessment)
-- Database name: pta

-- =====================================================
-- CORE USER MANAGEMENT
-- =====================================================

CREATE TABLE users (
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    full_name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    user_type ENUM('student','teacher','admin') NOT NULL,
    department VARCHAR(50),
    registration_number VARCHAR(50) UNIQUE,
    is_verified BOOLEAN DEFAULT FALSE,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_login TIMESTAMP NULL,
    profile_image VARCHAR(255),
    INDEX idx_email (email),
    INDEX idx_user_type (user_type),
    INDEX idx_department (department),
    INDEX idx_registration_number (registration_number)
);

Why this table exists
    Central identity store for all authenticated actors (student, teacher, admin).

Why each column exists

    user_id: internal primary key.

    full_name: display, reports.

    email: login + communication.

    password_hash: secure authentication.

    user_type: role-based access control.

    department: filtering, access rules.

    registration_number: academic identity.

    is_verified: email verification state.

    is_active: soft disable accounts.

    profile_image: UI personalization.

    created_at: audit.

    last_login: security tracking.

-- NEW: Email verification tokens

CREATE TABLE email_verification_tokens (
    token_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    token VARCHAR(64) UNIQUE NOT NULL,
    expires_at DATETIME NOT NULL,
    is_used BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_token (token),
    INDEX idx_expires (expires_at)
);

Why this table exists
    Manages email verification lifecycle separately from password reset.

Why each column exists

    token: one-time verification link.

    expires_at: invalidation window.

    is_used: replay protection.

CREATE TABLE password_reset_otps (
    otp_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    otp_code VARCHAR(10) NOT NULL,
    expires_at DATETIME NOT NULL,
    is_used BOOLEAN DEFAULT FALSE,
    request_ip VARCHAR(45),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_otp_code (otp_code),
    INDEX idx_user_expires (user_id, expires_at)
);

Why this table exists
    Implements OTP-based password recovery.

Why each column exists

    otp_code: sent to email.

    expires_at: OTP expiry.

    is_used: single-use enforcement.

    request_ip: abuse detection.

CREATE TABLE login_activity (
    log_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    ip_address VARCHAR(45),
    user_agent TEXT,
    is_success BOOLEAN,
    failure_reason VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_user_created (user_id, created_at),
    INDEX idx_ip_created (ip_address, created_at)
);

Why this table exists
    Detects suspicious logins and brute-force attempts.

Why each column exists

    ip_address: anomaly detection.

    user_agent: device fingerprinting.

    failure_reason: security analytics.

    created_at: timeline reconstruction.

-- =====================================================
-- ASSESSMENT MANAGEMENT
-- =====================================================

CREATE TABLE assessments (
    assessment_id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    created_by INT NOT NULL,
    category VARCHAR(50),
    difficulty ENUM('easy','medium','hard') DEFAULT 'medium',
    duration_minutes INT NOT NULL,
    total_marks INT NOT NULL,
    passing_marks INT NOT NULL,
    available_from DATETIME,
    available_until DATETIME,
    max_attempts INT DEFAULT 1,
    show_results_immediately BOOLEAN DEFAULT TRUE,
    show_correct_answers BOOLEAN DEFAULT TRUE,
    randomize_questions BOOLEAN DEFAULT FALSE,
    randomize_options BOOLEAN DEFAULT FALSE,
    questions_per_attempt INT, -- NULL means all questions
    is_public BOOLEAN DEFAULT FALSE,
    status ENUM('draft','active','archived','scheduled') DEFAULT 'draft',
    instructions TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (created_by) REFERENCES users(user_id),
    INDEX idx_status (status),
    INDEX idx_category (category),
    INDEX idx_created_by (created_by),
    INDEX idx_available_dates (available_from, available_until)
);

Why this table exists
    Defines a test configuration independent of attempts.

Why each column exists

    created_by: ownership.

    category: classification.

    difficulty: baseline difficulty.

    duration_minutes: timer.

    total_marks: scoring.

    passing_marks: result logic.

    available_from / until: scheduling.

    max_attempts: retry limits.

    randomize_*: fairness.

    questions_per_attempt: adaptive tests.

    is_public: guest access.

    status: lifecycle state.

-- NEW: Assessment access control (who can take which assessment)
CREATE TABLE assessment_access (
    access_id INT PRIMARY KEY AUTO_INCREMENT,
    assessment_id INT NOT NULL,
    user_id INT,
    department VARCHAR(50),
    registration_pattern VARCHAR(50),
    access_type ENUM('allow','deny') DEFAULT 'allow',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (assessment_id) REFERENCES assessments(assessment_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_assessment (assessment_id),
    INDEX idx_user (user_id)
);

Why this table exists
    Fine-grained control over who can access which assessment.

Why each column exists

    user_id: per-user allow/deny.

    department: batch access control.

    registration_pattern: regex-like filtering.

    access_type: allow vs deny.

CREATE TABLE questions (
    question_id INT PRIMARY KEY AUTO_INCREMENT,
    assessment_id INT NOT NULL,
    question_type ENUM('mcq','multiple_select','true_false','short_answer','fill_blank','match') NOT NULL,
    question_text TEXT NOT NULL,
    marks INT NOT NULL,
    negative_marks DECIMAL(4,2) DEFAULT 0,
    option_a TEXT,
    option_b TEXT,
    option_c TEXT,
    option_d TEXT,
    correct_answer TEXT NOT NULL,
    explanation TEXT,
    question_image VARCHAR(255),
    topic VARCHAR(100),
    question_order INT,
    difficulty ENUM('easy','medium','hard') DEFAULT 'medium',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (assessment_id) REFERENCES assessments(assessment_id) ON DELETE CASCADE,
    INDEX idx_assessment (assessment_id),
    INDEX idx_topic (topic),
    INDEX idx_order (assessment_id, question_order)
);

Why this table exists
    Stores all assessment questions, including PDF-parsed content.

Why each column exists

    question_type: evaluation logic.

    marks: scoring.

    negative_marks: penalty logic.

    option_*: MCQ support.

    correct_answer: auto-grading.

    explanation: learning feedback.

    question_image: media support.

    topic: analytics grouping.

    difficulty: per-question calibration.

    question_order: rendering order.

-- NEW: Track which questions were shown in each attempt (for randomization)
CREATE TABLE attempt_questions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    attempt_id INT NOT NULL,
    question_id INT NOT NULL,
    question_order INT NOT NULL,
    FOREIGN KEY (attempt_id) REFERENCES assessment_attempts(attempt_id) ON DELETE CASCADE,
    FOREIGN KEY (question_id) REFERENCES questions(question_id) ON DELETE CASCADE,
    UNIQUE KEY unique_attempt_question (attempt_id, question_id),
    INDEX idx_attempt (attempt_id)
);

Why this table exists
    Locks randomized question sets per attempt.

Why each column exists

    question_order: preserves order.

    unique_attempt_question: prevents duplication.


CREATE TABLE assessment_attempts (
    attempt_id INT PRIMARY KEY AUTO_INCREMENT,
    assessment_id INT NOT NULL,
    user_id INT,
    guest_name VARCHAR(100),
    guest_email VARCHAR(100),
    guest_token VARCHAR(64),
    attempt_number INT DEFAULT 1,
    start_time DATETIME,
    end_time DATETIME,
    submitted_at DATETIME,
    score DECIMAL(6,2),
    percentage DECIMAL(5,2),
    total_questions INT,
    correct_answers INT,
    wrong_answers INT,
    unanswered INT,
    status ENUM('in_progress','completed','abandoned','timeout','under_review') DEFAULT 'in_progress',
    ip_address VARCHAR(45),
    user_agent TEXT,
    browser_fingerprint VARCHAR(255), -- NEW: Additional security
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (assessment_id) REFERENCES assessments(assessment_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    INDEX idx_assessment (assessment_id),
    INDEX idx_user (user_id),
    INDEX idx_status (status),
    INDEX idx_guest_token (guest_token)
);

Why this table exists
    Represents one sitting of an assessment.

Why each column exists

    attempt_number: retry tracking.

    guest_*: guest support.

    guest_token: session resume.

    submitted_at: submission time.

    correct/wrong/unanswered: instant analytics.

    status: lifecycle state.

    browser_fingerprint: cheating detection.

-- NEW: Track overall system usage
CREATE TABLE system_analytics (
    stat_id INT PRIMARY KEY AUTO_INCREMENT,
    stat_date DATE NOT NULL,
    metric_name VARCHAR(100) NOT NULL,
    metric_value DECIMAL(15,2),
    metadata JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_date_metric (stat_date, metric_name),
    INDEX idx_date (stat_date)
);

Why this table exists
    Stores aggregated, system-wide metrics for dashboards and trend analysis without scanning large transactional tables.

Why each column exists

    stat_id: Internal primary key for efficient referencing and maintenance.

    stat_date: Date the metric applies to (supports daily aggregation and time-series analysis).

    metric_name: Identifier for the metric (e.g., daily_active_users, tests_taken, avg_score, new_registrations).

    metric_value: Numeric value of the metric (counts, averages, rates).

    metadata: Flexible JSON for contextual details (e.g., breakdown by role, category, device, or region).

    created_at: Audit timestamp indicating when the metric was recorded.

    unique_date_metric: Enforces one value per metric per day, preventing duplicates.

    idx_date: Optimizes date-range queries for charts and reports.
